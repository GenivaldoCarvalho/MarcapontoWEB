function menuToggle(){
    let menuArea = document.getElementById("menu-area");
    menuArea.classList.add('menu-opened')

    if(menuArea.classList.constains('menu-opened') == true)
    {
        menuArea.classList.remove('menu-opened');

    } else{
        menuArea.classList.add('menu-opened');
    }    

}
